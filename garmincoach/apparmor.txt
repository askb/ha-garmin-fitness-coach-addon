# AppArmor profile for GarminCoach HA Addon

#include <tunables/global>

profile garmincoach flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # S6-overlay init system
  /init rix,
  /run/** rwix,
  /command/** rix,
  /package/** r,
  /etc/s6-overlay/** r,
  /etc/s6-overlay/s6-rc.d/*/run rix,
  /etc/services.d/*/run rix,
  /etc/cont-init.d/* rix,

  # S6 binaries
  /bin/** rix,
  /usr/bin/** rix,
  /usr/local/bin/** rix,

  # Node.js runtime
  /app/** rmix,

  # Python for Garmin sync
  /usr/lib/python3*/** r,

  # Data directory (persistent storage)
  /data/** rwk,

  # Temp files
  /tmp/** rwk,

  # DNS resolution
  /etc/resolv.conf r,
  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/** r,

  # Network access (Garmin API, HA API, Ollama)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # Proc filesystem (Node.js needs this)
  /proc/** r,
  /sys/fs/cgroup/** r,

  # Libraries
  /lib/** rm,
  /usr/lib/** rm,

  # Device access
  /dev/null rw,
  /dev/urandom r,
  /dev/tty rw,
}
